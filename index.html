function doGet(e) { var action = e.parameter.action; var ss = SpreadsheetApp.getActiveSpreadsheet(); // 1. ACTION: Get Logs for the Dashboard if (action === 'getLogs') { var sheet = ss.getSheetByName('LOGS'); if (!sheet) return createJson({logs: []}); var data = sheet.getDataRange().getValues(); var logs = []; for (var i = 1; i < data.length; i++) { logs.push({ username: data[i][0], role: data[i][1], tab: data[i][2], loginDate: data[i][3], loginTime: data[i][4], logoutTime: data[i][5], deviceInfo: data[i][6] }); } return createJson({logs: logs.reverse()}); // Latest first } // 2. ACTION: Save Log (Login or Logout) if (action === 'saveLog') { var logData = JSON.parse(e.parameter.data); var sheet = ss.getSheetByName('LOGS'); if (!sheet) { sheet = ss.insertSheet('LOGS'); sheet.appendRow(['User', 'Role', 'Tab', 'Date', 'Login Time', 'Logout Time', 'Device']); } if (logData.action === 'logout') { // Find the last login row for this user and add logout time var data = sheet.getDataRange().getValues(); for (var i = data.length - 1; i >= 0; i--) { if (data[i][0] === logData.username && data[i][5] === '') { sheet.getRange(i + 1, 6).setValue(logData.logoutTime); break; } } } else { // It's a login sheet.appendRow([ logData.username, logData.role, logData.tab, logData.loginDate, logData.loginTime, '', logData.deviceInfo ]); } return ContentService.createTextOutput("Success"); } // 3. DEFAULT: Get Member Assignment Data var dataSheet = ss.getSheetByName('DATA'); // Make sure your data sheet is named DATA var rows = dataSheet.getDataRange().getValues(); return createJson({rows: rows}); } function createJson(obj) { return ContentService.createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); }
